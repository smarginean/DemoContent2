namespace Sdl.Connectors.MarketoService.Capabilities.Assets.Email.Contract.Content
{
    using Newtonsoft.Json;
    using Sdl.Connectors.MarketoService.Capabilities.Assets.Common.Contract;
    using System.Collections.Generic;

    public class EmailPreviewResponseEntity : BaseResponseEntity
    {
        [JsonIgnore]
        private List<EmailPreviewResultItem> result;

        public EmailPreviewResponseEntity()
        {
            this.Result = new List<EmailPreviewResultItem>();
        }

        /// <summary>Gets or sets the result.</summary>
        /// <value>The result.</value>
        [JsonProperty("result")]
        public List<EmailPreviewResultItem> Result
        {
            get
            {
                return this.result;
            }

            set
            {
                this.result = value;
            }
        }
    }
}
